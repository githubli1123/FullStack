Q6

模拟类

```

// 定义一个构造函数
function MyClass() {
    // 实例属性
    this.myProperty = 'Hello, World!';
}
 
// 在原型上定义方法
MyClass.prototype.myMethod = function() {
    console.log(this);
};
 
// 创建一个新的实例
var obj = new MyClass();
 
// 调用实例方法
obj.myMethod(); // 输出: Hello, World!
```
